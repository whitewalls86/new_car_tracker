# Plan 112 Gate A2: PySpark Iceberg fixture spike (lakehouse-worker image).
# Isolated from the shared FastAPI-services pip install, same convention as
# airflow/requirements.txt -- installing a JVM-backed engine alongside the
# rest of the repo's dependencies is not worth the resolver risk.
#
# pyspark is pinned to a version with a published matching
# iceberg-spark-runtime artifact (3.5_2.12) -- see lakehouse/Dockerfile for
# the exact Iceberg-Spark-runtime/iceberg-aws-bundle jar versions fetched at
# build time.
pyspark==3.5.3
boto3
requests

# Plan 112 Gate A3: reading the real int_listing_volatility_features table
# out of the analytics DuckDB file as a pandas DataFrame
# (scripts/export_volatility_features_to_iceberg.py, via duckdb's `.df()`)
# and handing it to spark.createDataFrame. Not needed by A2's in-process
# synthetic fixture. pyarrow is kept alongside pandas even though the script
# doesn't import it directly -- PySpark's createDataFrame(pandas_df) uses it
# for the Arrow-optimized conversion path when available.
duckdb
pyarrow
pandas
# duckdb's Python client requires pytz to convert timestamp columns (e.g.
# latest_fetched_at) even for a plain fetchone() -- not installed
# transitively by duckdb/pandas/pyarrow, confirmed missing on the first VM
# A3 run (_duckdb.InvalidInputException: Required module 'pytz' failed to
# import).
pytz

# Plan 125 Gate A: dbt-spark execution against the Lakekeeper REST catalog,
# in-process via `method: session` (no Thrift server -- keeps the one-shot
# worker shape this image already has). Pins are deliberate, not floating:
#
#   * dbt-core 1.10.20 matches the repo-wide pin (dbt/Dockerfile,
#     dbt_runner/Dockerfile, CI). Do not diverge -- the same dbt/ project
#     directory is compiled by both this image and dbt_runner.
#   * dbt-spark 1.10.3 is the latest 1.10.x. dbt-spark 1.11.0 released
#     2026-07-16 and has no track record here.
#   * The [session] extra pulls dbt-spark's PySpark integration; pyspark
#     itself stays pinned at 3.5.3 above (matching the 3.5_2.12 Iceberg
#     runtime jar), and the extra's `pyspark>=3.0,<4.0` range accepts it.
#
# This is installed ONLY in the lakehouse image -- never alongside
# dbt-duckdb/dbt-postgres in dbt_runner, per the same isolation convention
# that keeps airflow/requirements.txt separate (docs/plan_125_portability_audit.md,
# "Gate A adapter choice").
dbt-core==1.10.20
dbt-spark[session]==1.10.3

# Python 3.13 (this image's base -- see lakehouse/Dockerfile) removed
# distutils from the stdlib entirely. PySpark 3.5.3's
# require_minimum_pandas_version() still does
# `from distutils.version import LooseVersion`, which otherwise fails with
# ModuleNotFoundError (confirmed on the second VM A3 run, right after the
# DuckDB .df() fix). setuptools installs a `.pth`-based import shim
# (`distutils -> setuptools._distutils`) that's active for every subsequent
# Python process in the image, regardless of import order -- no code change
# needed, just this dependency.
setuptools
